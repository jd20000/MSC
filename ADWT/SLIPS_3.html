use clgDB // Departments Collection=id,name(dept),location db.departments.insertMany([ { _id: 1, name: "HR", location: "Mumbai" }, { _id: 2, name: "Sales", location: "Pune" }, { _id: 3, name: "IT", location: "Nashik" }, { _id: 4, name: "Accounts", location: "Nagpur" }, { _id: 5, name: "Marketing", location: "Thane" } ]); // Employees Collection=name,sal,dept db.employees.insertMany([ { name: "Ravi", salary: 45000, dept_id: 1 }, { name: "Sneha", salary: 55000, dept_id: 2 }, { name: "Amit", salary: 75000, dept_id: 3 }, { name: "Priya", salary: 60000, dept_id: 2 }, { name: "Kiran", salary: 40000, dept_id: 4 }, { name: "Rohit", salary: 50000, dept_id: 5 }, { name: "Neha", salary: 70000, dept_id: 3 } ]); // Queries // a. Display name of employee who has highest salary db.employees.find().sort({ salary: -1 }).limit(1) // b. Display biggest department with max. no. of employees print( "Department:", db.departments.findOne({ _id: db.employees.aggregate([ { $group: { _id: "$dept_id", c: { $sum: 1 } } }, { $sort: { c: -1 } }, { $limit: 1 } ]).toArray()[0]._id }).name ); // c. Write a cursor which shows department wise employee information db.departments.find().forEach(d => db.employees.find({ dept_id: d._id }).forEach(e => print(d.name, " ", e.name, "|", e.salary) ) ); // d. List all the employees who work in Sales dept and salary > 50000 db.employees.find({ dept_id: db.departments.findOne({ name: "Sales" })._id, salary: { $gt: 50000 } })
